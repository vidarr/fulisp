41 /* Advantage of THIS implementation:
42  * Single pointer derefencing to access car/cdr
43  * No second struct to wrap car/cdr
44  * 'Extended' types possible (i.e. some BIGNUM with head = mantisse,
45  * tail =
46  exponent
47  Disadvantages:
48  * Atomic types seize one int that is not used
49  */
50 struct Expression {
51  int type;
52  union {
53  char *string;
54  int integer;
55  char character;
56  double floating;
57  struct Expression *(*nativeFunc)(struct Expression *env, struct Expression *);
58  struct Environment *env;
59  struct Lambda *lambda;
60  struct Expression *car;
61  } data;
62  union {
63  struct Expression *cdr;
64  size_t length; /* Used for length of string */
65  } extension;
66  GC_DEFINE_EXPR_INFO
67 };
